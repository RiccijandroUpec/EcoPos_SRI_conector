// script.SriInvoiceOff
// --- ecopos-sri-connector hook -------------------------------------------
// DATA-ONLY resource - counterpart to script.SriInvoiceOn.txt. Turns
// electronic invoicing with the SRI OFF GLOBALLY (every sale from now on)
// until script.SriInvoiceOn is pressed again - the ticket still prints
// normally either way, this only controls whether Ticket.Close.xml drops
// the flag file ecopos-sri-connector watches for.

try {
    java.io.File dirname = new java.io.File(System.getProperty("dirname.path", "./"));
    java.io.File carpetaConector = new java.io.File(dirname, "sri-conector");
    carpetaConector.mkdirs();
    java.io.File archivoEstado = new java.io.File(carpetaConector, "facturacion-global.properties");
    java.util.Properties estado = new java.util.Properties();
    estado.setProperty("activo", "false");
    java.io.FileWriter escritor = new java.io.FileWriter(archivoEstado);
    estado.store(escritor, "ecopos-sri-connector - estado global de facturacion electronica");
    escritor.close();
} catch (Exception eEstado) {
    javax.swing.JOptionPane.showMessageDialog(null, "No se pudo guardar el estado global de facturación: " + eEstado.getMessage());
}

mensaje = new javax.swing.JLabel(
    "<html><div style='text-align:center;'>" +
    "<span style='font-size:16px;'>Facturación electrónica SRI</span><br>" +
    "<span style='font-size:24px;color:#a02020;'><b>DESACTIVADA</b></span><br>" +
    "<span style='font-size:11px;'>para TODAS las ventas, hasta que se active<br>(el ticket se imprime igual)</span>" +
    "</div></html>");
javax.swing.JOptionPane.showMessageDialog(null, mensaje, "Facturación electrónica SRI", javax.swing.JOptionPane.PLAIN_MESSAGE);
