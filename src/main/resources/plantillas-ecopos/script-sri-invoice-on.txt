// script.SriInvoiceOn
// --- ecopos-sri-connector hook -------------------------------------------
// DATA-ONLY resource - turns electronic invoicing with the SRI ON
// GLOBALLY (every sale from now on, not just this ticket) until
// script.SriInvoiceOff is pressed. Persisted in a small properties file
// next to the connector (sri-conector/facturacion-global.properties, key
// "activo"), read directly by Ticket.Close.xml - independent of any
// single ticket's lifecycle, so it survives across tickets/shifts without
// the cashier having to press it again per sale. Never affects ticket
// printing, which happens unconditionally regardless of this setting.
//
// Also makes sure there's an email to send THIS ticket's invoice to
// automatically once the SRI authorizes it. If the customer already
// selected on this ticket has one saved in their profile
// (CustomerInfoExt.getEmail()), that one is used as-is - no need to ask
// again. Only if there's no customer profile email (walk-in/consumidor
// final, or a customer without one saved) does it ask, storing the answer
// as the "sri.email" ticket property. Not required to proceed either way -
// leaving it blank just means no automatic email for this sale; it can
// still be sent later by hand from the connector's Historial screen.

try {
    java.io.File dirname = new java.io.File(System.getProperty("dirname.path", "./"));
    java.io.File carpetaConector = new java.io.File(dirname, "sri-conector");
    carpetaConector.mkdirs();
    java.io.File archivoEstado = new java.io.File(carpetaConector, "facturacion-global.properties");
    java.util.Properties estado = new java.util.Properties();
    estado.setProperty("activo", "true");
    java.io.FileWriter escritor = new java.io.FileWriter(archivoEstado);
    estado.store(escritor, "ecopos-sri-connector - estado global de facturacion electronica");
    escritor.close();
} catch (Exception eEstado) {
    javax.swing.JOptionPane.showMessageDialog(null, "No se pudo guardar el estado global de facturación: " + eEstado.getMessage());
}

correoDePerfil = null;
if (ticket.getCustomer() != null) {
    correoDePerfil = ticket.getCustomer().getEmail();
}
if (correoDePerfil == null || correoDePerfil.trim().equals("")) {
    correo = javax.swing.JOptionPane.showInputDialog(null,
        "Este cliente no tiene un correo guardado en su perfil.\n" +
        "Correo para enviarle ESTA factura electrónica (déjalo vacío si no aplica):",
        ticket.getProperty("sri.email", ""));
    if (correo != null) {
        ticket.setProperty("sri.email", correo.trim());
    }
}

mensaje = new javax.swing.JLabel(
    "<html><div style='text-align:center;'>" +
    "<span style='font-size:16px;'>Facturación electrónica SRI</span><br>" +
    "<span style='font-size:24px;color:#178a4c;'><b>ACTIVADA</b></span><br>" +
    "<span style='font-size:11px;'>para TODAS las ventas, hasta que se desactive</span>" +
    "</div></html>");
javax.swing.JOptionPane.showMessageDialog(null, mensaje, "Facturación electrónica SRI", javax.swing.JOptionPane.PLAIN_MESSAGE);
